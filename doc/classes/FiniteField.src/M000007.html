  def inverse(a)
    remainder = [0,0,0]
    quotient = [0,0,0]
    auxillary = [0,0,0]
    
    remainder[1] = @polynomial
    remainder[2] = a
    auxillary[1] = 0
    auxillary[2] = 1
    i = 2
    
    while(remainder[i] > 1)
      i += 1
      result = binary_div(remainder[i-2], remainder[i-1])
      remainder[i] = result[1]
      quotient[i] = result[0]
      auxillary[i] = binary_mul(quotient[i], auxillary[i-1]) ^ auxillary[i-2]
    end
    
    return auxillary[i]
  end
